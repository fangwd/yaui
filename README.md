## Install

```bash
npm install --save yaui
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'yaui'
import 'yaui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
